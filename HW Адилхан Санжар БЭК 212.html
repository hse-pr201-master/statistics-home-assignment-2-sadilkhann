    P = 1/n
    for i in range(1, k - 1):

import numpy as np
from scipy.stats import norm
np.random.seed(2004)

n_iter = 10000
sample_size = 20
intensity = 1
confidence_level = 0.95

def CI(sample, confidence_level):
    n = len(sample)
    mean = np.mean(sample)
    std = np.std(sample, ddof=1)
    z = norm.ppf((1 + confidence_level) / 2)
    left_b = mean - z * (std / np.sqrt(n))
    right_b = mean + z * (std / np.sqrt(n))
    return left_b, right_b

cnt= 0  
for i in range(n_iter):
    sample = np.random.exponential(scale = 1/intensity, size = sample_size)
    left_b, right_b = CI(sample, confidence_level)
    if left_b <= 1/intensity <= right_b:
        cnt += 1
    else:
        continue
round(cnt/n_iter, 3)

Исходы, при которых Вася получает выигрыш:
